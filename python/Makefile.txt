.PHONY: check lint type run clean

PYTHON := python
SRC := src

check: lint type
	@echo "✔ All checks passed"

lint:
	@echo "→ Ruff"
	ruff check $(SRC)

type:
	@echo "→ Mypy"
	mypy $(SRC)
	@echo "→ Pyright"
	pyright $(SRC)

run: check
	@echo "→ Running pipeline"
	$(PYTHON) $(SRC)/eebo_parse_tei.py

clean:
	find . -type d -name "__pycache__" -exec rm -rf {} +
